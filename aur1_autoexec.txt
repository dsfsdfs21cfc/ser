    local resp = req({ Url = "http://103.27.157.134:8452/time", Method = "GET" })
    if resp and resp.StatusCode == 200 then
        local serverTime = tonumber(resp.Body)
        if serverTime then
            _serverTimeOffset = serverTime - os.time()
        end
    end

local function isPlayerCharacter(model)
    return Players:GetPlayerFromCharacter(model) ~= nil
end

local function handleAnimator(animator)
    local model = animator:FindFirstAncestorOfClass("Model")
    if model and isPlayerCharacter(model) then return end
    for _, track in pairs(animator:GetPlayingAnimationTracks()) do track:Stop(0) end
    animator.AnimationPlayed:Connect(function(track) track:Stop(0) end)
end

local function stripVisuals(obj)
    local model = obj:FindFirstAncestorOfClass("Model")
    local isPlayer = model and isPlayerCharacter(model)

    if obj:IsA("Animator") then handleAnimator(obj) end

    if obj:IsA("Accessory") or obj:IsA("Clothing") then
        if obj:FindFirstAncestorOfClass("Model") then
            obj:Destroy()
        end
    end

    if not isPlayer then
        if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or
           obj:IsA("Smoke") or obj:IsA("Fire") or obj:IsA("Sparkles") or
           obj:IsA("Highlight") then
            obj.Enabled = false
        end
        if obj:IsA("Explosion") then
            obj:Destroy()
        end
        if obj:IsA("MeshPart") then
            obj.TextureID = ""
        end
    end

    if obj:IsA("BasePart") then
        obj.Material = Enum.Material.Plastic
        obj.Reflectance = 0
        obj.CastShadow = false
    end

    if obj:IsA("SurfaceAppearance") or obj:IsA("Texture") or obj:IsA("Decal") then
        obj:Destroy()
    end
end

-- Cash reader
local currentCash = 0
local function getCashString()
    local val = 0
    pcall(function()
        local ls = localPlayer:FindFirstChild("leaderstats")
        if ls then
            local cash = ls:FindFirstChild("Cash")
            if cash then
                local str = tostring(cash.Value):gsub("%$",""):gsub(",",""):gsub(" ","")
                local num, suffix = str:match("([%d%.]+)(%a*)")
                num = tonumber(num) or 0
                suffix = (suffix or ""):upper()
                if suffix == "K" then num = num * 1e3
                elseif suffix == "M" then num = num * 1e6
                elseif suffix == "B" then num = num * 1e9
                elseif suffix == "T" then num = num * 1e12
                elseif suffix == "QA" then num = num * 1e15
                elseif suffix == "QI" then num = num * 1e18
                end
                currentCash = num
            end
        end
    end)
    if currentCash >= 1e18 then return string.format("%.1fQI", currentCash / 1e18)
    elseif currentCash >= 1e15 then return string.format("%.1fQA", currentCash / 1e15)
    elseif currentCash >= 1e12 then return string.format("%.1fT", currentCash / 1e12)
    elseif currentCash >= 1e9 then return string.format("%.1fB", currentCash / 1e9)
    elseif currentCash >= 1e6 then return string.format("%.1fM", currentCash / 1e6)
    elseif currentCash >= 1e3 then return string.format("%.1fK", currentCash / 1e3)
    end
    return string.format("%.0f", currentCash)
end

local bxor = bit32 and bit32.bxor or bit.bxor
local band = bit32 and bit32.band or bit.band
local rshift = bit32 and bit32.rshift or bit.rshift
local lshift = bit32 and bit32.lshift or bit.lshift

local function mixHash(a, b, c, d)
    a = band(a + b, 0xFFFFFFFF); d = bxor(d, a); d = bxor(lshift(band(d, 0xFFFF), 16), rshift(d, 16))
    c = band(c + d, 0xFFFFFFFF); b = bxor(b, c); b = bxor(lshift(band(b, 0xFFFFF), 12), rshift(b, 20))
    a = band(a + b, 0xFFFFFFFF); d = bxor(d, a); d = bxor(lshift(band(d, 0xFFFFFF), 8), rshift(d, 24))
    c = band(c + d, 0xFFFFFFFF); b = bxor(b, c); b = bxor(lshift(band(b, 0x1FFFFFF), 7), rshift(b, 25))
    return a, b, c, d
end

local function generateKeyStream(iv, length)
    local s1, s2, s3, s4 = 0x6A09E667, 0xBB67AE85, 0x3C6EF372, 0xA54FF53A
    for i = 1, #SECRET do s1 = bxor(s1, string.byte(SECRET, i) * 16777619); s1, s2, s3, s4 = mixHash(s1, s2, s3, s4) end
    for i = 1, #iv do s2 = bxor(s2, string.byte(iv, i) * 16777619); s1, s2, s3, s4 = mixHash(s1, s2, s3, s4) end
    local stream, generated = {}, 0
    while generated < length do
        s1, s2, s3, s4 = mixHash(s1, s2, s3, s4)
        for _, s in ipairs({s1, s2, s3, s4}) do
            for shift = 0, 24, 8 do
                if generated >= length then break end
                stream[#stream + 1] = band(rshift(s, shift), 0xFF)
                generated = generated + 1
            end
        end
    end
    return stream
end

local function toHex(str)
    local t = {}
    for i = 1, #str do t[i] = string.format("%02x", string.byte(str, i)) end
    return table.concat(t)
end

local function EncryptJobId(jobId)
    local iv = {}
    for i = 1, 16 do iv[i] = string.char(math.random(0, 255)) end
    iv = table.concat(iv)
    local keyStream = generateKeyStream(iv, #jobId)
    local encrypted = {}
    for i = 1, #jobId do encrypted[i] = string.char(bxor(string.byte(jobId, i), keyStream[i])) end
    return toHex(iv .. table.concat(encrypted))
end

local function formatIncomeNumber(n)
    if not n then return "?" end
    if n >= 1e9 then
        local v = n / 1e9
        return (v % 1 == 0 and string.format("%dB/s", v) or string.format("%.1fB/s", v)):gsub("%.0B/s", "B/s")
    elseif n >= 1e6 then
        local v = n / 1e6
        return (v % 1 == 0 and string.format("%dM/s", v) or string.format("%.1fM/s", v)):gsub("%.0M/s", "M/s")
    elseif n >= 1e3 then
        local v = n / 1e3
        return (v % 1 == 0 and string.format("%dK/s", v) or string.format("%.1fK/s", v)):gsub("%.0K/s", "K/s")
    end
    return string.format("%d/s", n)
end

local function parseGenerationText(s)
    if type(s) ~= "string" or s == "" then return nil end
    local norm = s:gsub("%$", ""):gsub(",", ""):gsub("%s+", "")
    local num, suffix = norm:match("^([%-%d%.]+)([KkMmBb]?)/s$")
    if not num then return nil end
    local val = tonumber(num)
    if not val then return nil end
    local mult = { K = 1e3, k = 1e3, M = 1e6, m = 1e6, B = 1e9, b = 1e9 }
    return val * (mult[suffix] or 1)
end

local function grabText(inst)
    if not inst then return nil end
    if inst:IsA("TextLabel") or inst:IsA("TextButton") or inst:IsA("TextBox") then
        local ok, ct = pcall(function() return inst.ContentText end)
        if ok and type(ct) == "string" and #ct > 0 then return ct end
        local t = inst.Text
        if type(t) == "string" and #t > 0 then return t end
    end
    if inst:IsA("StringValue") then
        local v = inst.Value
        if type(v) == "string" and #v > 0 then return v end
    end
    return nil
end

local function isGuidName(s)
    return s:match("^[0-9a-fA-F]+%-%x+%-%x+%-%x+%-%x+$") ~= nil
end

local function getOverheadInfo(overhead)
    if not overhead then return nil, nil end
    local name
    local display = overhead:FindFirstChild("DisplayName")
    if display then name = grabText(display) end
    if not name then
        local any = overhead:FindFirstChildOfClass("TextLabel")
            or overhead:FindFirstChildOfClass("TextButton")
            or overhead:FindFirstChildOfClass("TextBox")
        name = any and grabText(any) or nil
    end
    local genText
    local gen = overhead:FindFirstChild("Generation")
    if gen then genText = grabText(gen) end
    if not genText then
        for _, child in ipairs(overhead:GetDescendants()) do
            if child:IsA("TextLabel") or child:IsA("TextButton") or child:IsA("TextBox") then
                local t = grabText(child)
                if t and (t:match("%$") or t:match("/s")) then
                    genText = t
                    break
                end
            end
        end
    end
    return name, genText
end

local function parseMutationFromOverhead(oh)
    if not oh then return nil end
    local mutLabel = oh:FindFirstChild("Mutation")
    if mutLabel and mutLabel:IsA("TextLabel") then
        local raw = mutLabel.Text or ""
        local clean = raw:gsub("<[^>]+>", ""):gsub("%s+", " "):match("^%s*(.-)%s*$") or ""
        if clean == "" or clean == "None" then return nil end
        -- DefaultState=true OR Visible=true means real mutation
        local ds = nil
        pcall(function() ds = mutLabel:GetAttribute("DefaultState") end)
        if ds == true or mutLabel.Visible then
            return clean
        end
    end
    return nil
end

-- Duel detection via player attribute __duels_block_steal
-- + RemainingTime on plot for cooldown timer
local function getPlotOwnerPlayer(plot)
    if not plot then return nil end
    local sign = plot:FindFirstChild("PlotSign", true)
    if not sign then return nil end
    -- PlotSign itself or its descendants may have the text
    local function checkText(inst)
        if inst:IsA("TextLabel") then
            local ownerName = inst.Text:match("^(.+)'s Base$")
            if ownerName and ownerName ~= "Empty" and ownerName ~= "" then
                for _, p in ipairs(Players:GetPlayers()) do
                    if p.DisplayName == ownerName or p.Name == ownerName then
                        return p
                    end
                end
            end
        end
        return nil
    end
    -- Check sign itself and descendants
    local found = checkText(sign)
    if found then return found end
    for _, desc in ipairs(sign:GetDescendants()) do
        found = checkText(desc)
        if found then return found end
    end
    -- Also check parent of sign (SurfaceGui > Frame > TextLabel)
    if sign.Parent then
        for _, desc in ipairs(sign.Parent:GetDescendants()) do
            found = checkText(desc)
            if found then return found end
        end
    end
    return nil
end

local function isPlotInDuel(plot)
    local owner = getPlotOwnerPlayer(plot)
    if not owner then return false, nil end
    local ok, attr = pcall(function() return owner:GetAttribute("__duels_block_steal") end)
    if ok and attr then
        -- Get timer from RemainingTime on plot
        local timer = nil
        for _, desc in ipairs(plot:GetDescendants()) do
            if desc:IsA("TextLabel") and desc.Name == "RemainingTime" then
                local secs = desc.Text:match("(%d+)")
                if secs then timer = tonumber(secs) end
                break
            end
        end
        return true, timer
    end
    return false, nil
end

local function isInDuel(oh)
    if not oh then return false end
    -- Active duel = visible "inDuel" label/text over the brainrot overhead.
    -- The label may always exist in overhead but only become visible during duel,
    -- so we require Visible=true AND TextTransparency<1 AND non-empty text.
    local function isLabelVisible(lbl)
        if lbl.Visible == false then return false end
        local ok, tt = pcall(function() return lbl.TextTransparency end)
        if ok and type(tt) == "number" and tt >= 1 then return false end
        local t = lbl.Text
        if type(t) ~= "string" or #t == 0 then return false end
        return true
    end
    for _, desc in ipairs(oh:GetDescendants()) do
        if desc:IsA("TextLabel") or desc:IsA("TextButton") or desc:IsA("TextBox") then
            local nm = desc.Name or ""
            if nm == "inDuel" or nm == "InDuel" or nm == "Induel" then
                if isLabelVisible(desc) then return true end
            else
                local t = desc.Text
                if type(t) == "string" and #t > 0 then
                    local low = t:lower():gsub("%s+", "")
                    if low:find("induel", 1, true) and isLabelVisible(desc) then
                        return true
                    end
                end
            end
        end
    end
    return false
end

local function isDuelCooldownBrainrot(oh, plot)
    return isPlotInDuel(plot)
end

-- Find plot by position (for brainrots in Debris that aren't inside a plot)
local function findPlotByPosition(pos)
    if not pos then return nil end
    local Plots = Workspace:FindFirstChild("Plots")
    if not Plots then return nil end
    for _, plot in ipairs(Plots:GetChildren()) do
        local pods = plot:FindFirstChild("AnimalPodiums")
        if pods then
            for _, pod in ipairs(pods:GetChildren()) do
                local base = pod:FindFirstChild("Base")
                local spawn = base and base:FindFirstChild("Spawn")
                if spawn and spawn:IsA("BasePart") then
                    if (spawn.Position - pos).Magnitude < 15 then
                        return plot
                    end
                end
            end
        end
    end
    return nil
end

-- Check duel status by position
local function isDuelByPosition(pos)
    local plot = findPlotByPosition(pos)
    if not plot then return false, nil end
    return isPlotInDuel(plot)
end

-- Filter out brainrots teleported off-map (Y > 1000 or Y < -1000 = off-map bug)
local function isValidPosition(pos)
    if not pos then return true end
    -- Map bounds: X(-536..-287), Z(-141..258) + margin 50
    if pos.X < -590 or pos.X > -230 or pos.Z < -200 or pos.Z > 310 then
        return false
    end
    return true
end

local function scanPlots()
    local r = {}
    local Plots = Workspace:FindFirstChild("Plots")
    if not Plots then return r end

    -- ── Primary: Synchronizer (server data, never misses) ──
    if _syncReady then
        for _, plot in ipairs(Plots:GetChildren()) do
            pcall(function()
                local ch = _Synchronizer:Get(plot.Name)
                if not ch then return end
                local al = ch:Get("AnimalList")
                if not al then return end

                -- Get owner
                local ownerName = "Unknown"
                local ownerPlayer = nil
                local owner = ch:Get("Owner")
                if owner then
                    if typeof(owner) == "Instance" and owner:IsA("Player") then
                        ownerPlayer = owner
                        ownerName = owner.DisplayName or owner.Name
                    elseif type(owner) == "table" and owner.Name then
                        ownerName = owner.Name
                        ownerPlayer = Players:FindFirstChild(owner.Name)
                    end
                end
                -- Duel cooldown (from player attribute)
                local cooldown, duelTimer = false, nil
                if ownerPlayer then
                    local ok, attr = pcall(function() return ownerPlayer:GetAttribute("__duels_block_steal") end)
                    if ok and attr then
                        cooldown = true
                        for _, desc in ipairs(plot:GetDescendants()) do
                            if desc:IsA("TextLabel") and desc.Name == "RemainingTime" then
                                local secs = desc.Text:match("(%d+)")
                                if secs then duelTimer = tonumber(secs) end
                                break
                            end
                        end
                    end
                end

                for slot, ad in pairs(al) do
                    if type(ad) == "table" and ad.Index then
                        local aInfo = _AnimalsData[ad.Index]
                        if aInfo then
                            local name = aInfo.DisplayName or ad.Index
                            local gv = _AnimalsShared:GetGeneration(ad.Index, ad.Mutation, ad.Traits, nil)
                            local mut = ad.Mutation
                            if mut == "None" or mut == "" then mut = nil end

                            -- Active duel: check overhead if it exists
                            local activeDuel = false
                            local podiums = plot:FindFirstChild("AnimalPodiums")
                            if podiums then
                                local podium = podiums:FindFirstChild(tostring(slot))
                                if podium then
                                    local oh = podium:FindFirstChild("AnimalOverhead", true)
                                    if oh then activeDuel = isInDuel(oh) end
                                end
                            end

                            local uid = plot.Name .. "_" .. tostring(slot)
                            table.insert(r, {
                                name = name, gen = gv, location = "Plot",
                                mutation = mut, inDuel = activeDuel,
                                duelCooldown = cooldown, duelTimer = duelTimer,
                                plotOwner = ownerName, uid = uid,
                            })
                        end
                    end
                end
            end)
        end
        return r
    end

    -- ── Fallback: visual scan (if Synchronizer failed to load) ──
    for _, plot in ipairs(Plots:GetChildren()) do
        local Podiums = plot:FindFirstChild("AnimalPodiums")
        if Podiums then
            for _, podium in ipairs(Podiums:GetChildren()) do
                local Base = podium:FindFirstChild("Base")
                local Spawn = Base and Base:FindFirstChild("Spawn")
                local Att = Spawn and Spawn:FindFirstChild("Attachment")
                local OH = Att and Att:FindFirstChild("AnimalOverhead")
                if OH then
                    local name, gt = getOverheadInfo(OH)
                    local gn = gt and parseGenerationText(gt)
                    if name and gn then
                        local mutation = parseMutationFromOverhead(OH)
                        local activeDuel = isInDuel(OH)
                        local cooldown, duelTimer = isPlotInDuel(plot)
                        local ownerPlayer = getPlotOwnerPlayer(plot)
                        local ownerName = ownerPlayer and ownerPlayer.DisplayName or "Unknown"
                        local pos = Spawn and Spawn:IsA("BasePart") and Spawn.Position or nil
                        local uid = name .. ":" .. tostring(gn) .. ":" .. (pos and string.format("%.0f,%.0f,%.0f", pos.X, pos.Y, pos.Z) or tostring(podium))
                        table.insert(r, { name = name, gen = gn, location = "Plot", mutation = mutation, inDuel = activeDuel, duelCooldown = cooldown, duelTimer = duelTimer, plotOwner = ownerName, uid = uid })
                    end
                end
            end
        end
    end
    return r
end

local function scanRunway()
    local r = {}
    for _, obj in ipairs(Workspace:GetChildren()) do
        if isGuidName(obj.Name) then
            -- Try standard path: Part > Info > AnimalOverhead
            local oh = nil
            local part = obj:FindFirstChild("Part")
            if part then
                local info = part:FindFirstChild("Info")
                oh = info and info:FindFirstChild("AnimalOverhead")
                -- Also try directly under Part
                if not oh then
                    oh = part:FindFirstChild("AnimalOverhead")
                end
            end
            -- Fallback: recursive search in the GUID model
            if not oh then
                oh = obj:FindFirstChild("AnimalOverhead", true)
            end
            if oh then
                -- Check if nearest plot has an owner in the server
                -- No nearby plot = ghost brainrot (owner left), skip it
                local pos = nil
                local basePart = obj:FindFirstChildWhichIsA("BasePart", true)
                if basePart then pos = basePart.Position end
                if not pos then continue end
                local nearestPlot = findPlotByPosition(pos)
                if not nearestPlot then continue end

                local ownerInServer = false
                if _syncReady then
                    pcall(function()
                        local ch = _Synchronizer:Get(nearestPlot.Name)
                        if ch then
                            local owner = ch:Get("Owner")
                            if owner then
                                if typeof(owner) == "Instance" and owner:IsA("Player") and owner.Parent then
                                    ownerInServer = true
                                elseif type(owner) == "table" and owner.Name and Players:FindFirstChild(owner.Name) then
                                    ownerInServer = true
                                end
                            end
                        end
                    end)
                else
                    local ownerPlayer = getPlotOwnerPlayer(nearestPlot)
                    ownerInServer = ownerPlayer ~= nil
                end
                if not ownerInServer then continue end

                local name, gt = getOverheadInfo(oh)
                local gn = gt and parseGenerationText(gt)
                if name and gn then
                    local mutation = parseMutationFromOverhead(oh)
                    local duel = isInDuel(oh)
                    local uid = "runway:" .. obj.Name
                    table.insert(r, { name = name, gen = gn, location = "Runway", mutation = mutation, inDuel = duel, uid = uid })
                end
            end
        end
    end
    return r
end

local function scanAllOverheads()
    local r = {}
    local seen = {}
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj.Name == "AnimalOverhead" then
            -- Skip objects inside Plots (scanPlots covers them)
            local inPlots = obj:FindFirstAncestor("Plots")
            -- Skip objects inside Debris (scanDebrisFolder covers them)
            local inDebris = obj:FindFirstAncestor("Debris")
            if not inPlots and not inDebris then
                local name, gt = getOverheadInfo(obj)
                local gn = gt and parseGenerationText(gt)
                if name and gn then
                    local key = name .. ":" .. tostring(gn)
                    if not seen[key] then
                        seen[key] = true
                        local pos = nil
                        local ancestor = obj:FindFirstAncestorOfClass("BasePart")
                        if ancestor then pos = ancestor.Position end
                        -- No nearby plot = ghost brainrot, skip
                        local nearestPlot = pos and findPlotByPosition(pos) or nil
                        if nearestPlot then
                            local ownerPlayer = getPlotOwnerPlayer(nearestPlot)
                            if ownerPlayer then
                                local mutation = parseMutationFromOverhead(obj)
                                local activeDuel = isInDuel(obj)
                                local uid = "world:" .. tostring(obj) .. ":" .. (pos and string.format("%.0f,%.0f,%.0f", pos.X, pos.Y, pos.Z) or "")
                                table.insert(r, { name = name, gen = gn, location = "World", mutation = mutation, inDuel = activeDuel, uid = uid })
                            end
                        end
                    end
                end
            end
        end
    end
    return r
end

local function scanPlayerGui()
    local r = {}
    local pg = localPlayer:FindFirstChild("PlayerGui")
    if not pg then return r end
    for _, child in ipairs(pg:GetDescendants()) do
        if child.Name == "AnimalOverhead" then
            local name, gt = getOverheadInfo(child)
            local gn = gt and parseGenerationText(gt)
            if name and gn then
                local mutation = parseMutationFromOverhead(child)
                local duel = isInDuel(child)
                local uid = "gui:" .. tostring(child)
                table.insert(r, { name = name, gen = gn, location = "GUI", mutation = mutation, inDuel = duel, uid = uid })
            end
        end
    end
    return r
end

local function scanDebrisFolder()
    local r = {}
    local df = Workspace:FindFirstChild("Debris")
    if not df then return r end
    for _, inst in ipairs(df:GetChildren()) do
        -- Scan all children, not just FastOverheadTemplate
        -- Filter fakes: if Motor6D exists but Part1 gone = brainrot left
        local motor = inst:FindFirstChild("__foh_transform")
        if not (motor and motor:IsA("Motor6D") and (not motor.Part1 or not motor.Part1.Parent)) then
            local oh = inst:FindFirstChild("AnimalOverhead")
            if not oh then
                local gui = inst:FindFirstChild("GUI")
                if gui then
                    oh = gui:FindFirstChild("AnimalOverhead") or gui
                end
            end
            -- Also search deeper if not found at top level
            if not oh then
                oh = inst:FindFirstChild("AnimalOverhead", true)
            end
            if oh then
                local ni = oh:FindFirstChild("DisplayName")
                local gi = oh:FindFirstChild("Generation")
                local name = ni and grabText(ni)
                local gt = gi and grabText(gi)
                if not name or not gt then
                    -- Fallback: try getOverheadInfo which checks more patterns
                    name, gt = getOverheadInfo(oh)
                end
                local gn = gt and parseGenerationText(gt)
                if name and gn then
                    local pos = inst:IsA("BasePart") and inst.Position or nil
                    -- No nearby plot = ghost brainrot, skip
                    local nearestPlot = pos and findPlotByPosition(pos) or nil
                    if nearestPlot then
                        local ownerPlayer = getPlotOwnerPlayer(nearestPlot)
                        if ownerPlayer then
                            local mutation = parseMutationFromOverhead(oh)
                            local activeDuel = isInDuel(oh)
                            local cooldown, duelTimer = isDuelByPosition(pos)
                            local uid = "debris:" .. tostring(inst) .. ":" .. (pos and string.format("%.0f,%.0f,%.0f", pos.X, pos.Y, pos.Z) or "")
                            table.insert(r, { name = name, gen = gn, location = "DebrisFolder", mutation = mutation, inDuel = activeDuel, duelCooldown = cooldown, duelTimer = duelTimer, uid = uid })
                        end
                    end
                end
            end
        end
    end
    return r
end

local function collectAll(timeout)
    local t0 = os.clock()
    local collected = {}
    repeat
        collected = {}
        local sources = {}
        -- Run all scans in parallel
        local done = 0
        local total = _syncReady and 2 or 5
        task.spawn(function() local r = scanPlots()          for _, v in ipairs(r) do table.insert(sources, v) end done += 1 end)
        task.spawn(function() local r = scanRunway()         for _, v in ipairs(r) do table.insert(sources, v) end done += 1 end)
        if not _syncReady then
            -- Fallback scanners only when Synchronizer unavailable
            task.spawn(function() local r = scanPlayerGui()      for _, v in ipairs(r) do table.insert(sources, v) end done += 1 end)
            task.spawn(function() local r = scanDebrisFolder()   for _, v in ipairs(r) do table.insert(sources, v) end done += 1 end)
            task.spawn(function() local r = scanAllOverheads()   for _, v in ipairs(r) do table.insert(sources, v) end done += 1 end)
        end
        while done < total and os.clock() - t0 < timeout do task.wait() end
        collected = sources
        -- Deduplicate: same name+gen+mutation = same brainrot found by multiple scanners
        -- But allow multiple if different gen or different mutation (truly different brainrots)
        local counts = {}  -- track how many of each name+gen+mutation combo we expect
        local seen = {}
        local unique = {}

        -- First pass: count unique positions per name+gen+mutation from Plot scanner (most reliable)
        local plotCounts = {}
        for _, item in ipairs(collected) do
            if item.location == "Plot" then
                local k = item.name .. ":" .. tostring(item.gen) .. ":" .. (item.mutation or "")
                plotCounts[k] = (plotCounts[k] or 0) + 1
            end
        end

        -- Second pass: prefer Plot > World > Debris
        local added = {}
        -- Add all Plot items (each is a unique brainrot on a unique podium)
        for _, item in ipairs(collected) do
            if item.location == "Plot" then
                table.insert(unique, item)
                local k = item.name .. ":" .. tostring(item.gen) .. ":" .. (item.mutation or "")
                added[k] = (added[k] or 0) + 1
            end
        end
        -- Then add non-Plot items only if Plot didn't find any with same name+gen+mutation
        for _, item in ipairs(collected) do
            if item.location ~= "Plot" then
                local k = item.name .. ":" .. tostring(item.gen) .. ":" .. (item.mutation or "")
                if not added[k] or added[k] == 0 then
                    table.insert(unique, item)
                    added[k] = 1
                end
            end
        end

        collected = unique
        if #collected > 0 then break end
        task.wait(0.1)
    until os.clock() - t0 > timeout
    return collected
end

local MIN_INCOME_OVERRIDE = {
    ["Guest 666"]   = 0,
    ["Noo my Eggs"] = 0,
}

-- Brainrots that skip lowrange (they have their own tier via MIN_INCOME_OVERRIDE)
local LOWRANGE_EXCLUDE = {
    ["Guest 666"]   = true,
    ["Noo my Eggs"] = true,
}

local function shouldShow(name, gen)
    if type(gen) ~= "number" then return false end
    local minIncome = MIN_INCOME_OVERRIDE[name]
    if minIncome then return gen >= minIncome end
    return gen >= 1_000_000
end

local function isLowRange(name, gen)
    if LOWRANGE_EXCLUDE[name] then return false end
    if MIN_INCOME_OVERRIDE[name] then return false end
    return type(gen) == "number" and gen >= 1_000_000 and gen < 25_000_000
end

local function isSpecialBrainrot(name, gen, mutation)
    return false
end

-- Simple key: name + gen + mutation (same for all scanners)
local function generateSimpleKey(obj)
    return obj.name .. ":" .. tostring(obj.gen)
end

-- Full key: includes rounded position (for tracking unique brainrots at different locations)
local function generateCacheKey(obj)
    local posStr = ""
    if obj.uid then
        local x, y, z = obj.uid:match("([%d%-%.]+),([%d%-%.]+),([%d%-%.]+)$")
        if x and y and z then
            posStr = tostring(math.floor(tonumber(x)/5)*5) .. "," .. tostring(math.floor(tonumber(y)/5)*5) .. "," .. tostring(math.floor(tonumber(z)/5)*5)
        end
    end
    return obj.name .. ":" .. posStr .. ":" .. tostring(obj.gen) .. ":" .. (obj.mutation or "none")
end

local function deduplicateForSend(objects)
    local seen = {}
    local unique = {}
    for _, obj in ipairs(objects) do
        local key = generateSimpleKey(obj)
        if not seen[key] then
            seen[key] = true
            table.insert(unique, obj)
        end
    end
    return unique
end

local function filterNewObjects(objects)
    -- First deduplicate by simple key (name+gen) — prefer version with mutation
    local deduped = {}
    local seenSimple = {}
    local seenIndex = {}
    for _, obj in ipairs(objects) do
        local sk = generateSimpleKey(obj)
        if not seenSimple[sk] then
            seenSimple[sk] = true
            seenIndex[sk] = #deduped + 1
            table.insert(deduped, obj)
        elseif obj.mutation and not deduped[seenIndex[sk]].mutation then
            -- Replace with version that has mutation
            deduped[seenIndex[sk]] = obj
        end
    end
    -- Then filter by sentCache using simple key with timestamp
    local now = os.clock()
    local r = {}
    for _, obj in ipairs(deduped) do
        local key = generateSimpleKey(obj)
        local lastSent = sentCache[key]
        if not lastSent or (now - lastSent) >= 90 then
            sentCache[key] = now
            table.insert(r, obj)
        end
    end
    return r
end

-- ── Persistent WebSocket to VDS ─────────────────────────
local _vds_ws = nil
local _vds_ws_ready = false

local function ConnectVDSWS()
    task.spawn(function()
        while true do
            local ok, ws = pcall(function()
                return WebSocket.connect(VDS_WS_URL)
            end)
            if ok and ws then
                _vds_ws = ws
                pcall(function()
                    ws:Send(HttpService:JSONEncode({password = VDS_SEND_PASSWORD}))
                end)
                _vds_ws_ready = true

                local closed = false
                ws.OnClose:Connect(function()
                    closed = true
                    _vds_ws_ready = false
                    _vds_ws = nil
                end)

                -- Ping every 60 sec to keep alive
                task.spawn(function()
                    while not closed do
                        task.wait(60)
                        if not closed and _vds_ws then
                            pcall(function() _vds_ws:Send(HttpService:JSONEncode({type = "ping"})) end)
                        end
                    end
                end)

                while not closed do task.wait(1) end
            end
            _vds_ws_ready = false
            _vds_ws = nil
            task.wait(3)
        end
    end)
end

if WebSocket and WebSocket.connect then
    ConnectVDSWS()
end

local function sendToVDS(filtered, wh, duelItems)
    if #filtered == 0 then return end
    local payload = {
        jobId = EncryptJobId(tostring(game.JobId)),
        placeId = game.PlaceId,
        title = wh.title,
        tier = wh.title:lower():gsub(" range", "range"),
        color = wh.color,
        range = { min = wh.min, max = wh.max },
        sendServerInfo = wh.sendServerInfo or false,
        time = getRealUTC(),
        bot = BOT_TAG,
        objects = {},
    }
    for _, obj in ipairs(filtered) do
        table.insert(payload.objects, {
            name = obj.name,
            gen = obj.gen,
            location = obj.location,
            mutation = obj.mutation or nil,
            important = OBJECTS[obj.name] or false,
            isSpecial = isSpecialBrainrot(obj.name, obj.gen, obj.mutation),
            inDuel = obj.inDuel or obj.duelCooldown or false,
        })
    end
    if duelItems then
        for _, obj in ipairs(duelItems) do
            table.insert(payload.objects, {
                name = obj.name,
                gen = obj.gen,
                location = obj.location,
                mutation = obj.mutation or nil,
                important = OBJECTS[obj.name] or false,
                isSpecial = isSpecialBrainrot(obj.name, obj.gen, obj.mutation),
                inDuel = true,
            })
        end
    end

    -- Try WebSocket first (instant)
    if _vds_ws_ready and _vds_ws then
        local sent = pcall(function()
            _vds_ws:Send(HttpService:JSONEncode({type = "brainrot", data = payload}))
        end)
        if sent then return end
    end

    -- Fallback: HTTP (in case WS not available or disconnected)
    local req = getRequester()
    if not req then return end
    pcall(function()
        return req({
            Url = VDS_URL .. "/brainrot",
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json",
                ["X-Aurora-Role"] = "send",
            },
            Body = HttpService:JSONEncode(payload),
        })
    end)
end

local function nameToSlug(name)
    return name:lower():gsub("%s+", "-")
end

local function getBrainrotImageUrl(name)
    return "https://steal-a-brainrot.org/images/brainrots/" .. nameToSlug(name) .. ".webp"
end

local function buildDiscordEmbed(filtered, wh, extraItems, extraTitle, duelItems)
    local jobId = game.JobId
    local placeId = game.PlaceId

    -- Sort all by gen descending
    local sorted = {}
    for _, obj in ipairs(filtered) do table.insert(sorted, obj) end
    table.sort(sorted, function(a, b) return a.gen > b.gen end)

    -- Priority brainrots for thumbnail image (kingrange)
    local PRIORITY_THUMBNAIL = {
        ["Meowl"] = true,
        ["Strawberry Elephant"] = true,
        ["Skibidi Toilet"] = true,
        ["Headless Horseman"] = true,
    }

    -- Build description
    local desc = ""

    -- Build duel lookup
    local duelLookup = {}
    if duelItems then
        for _, obj in ipairs(duelItems) do
            duelLookup[obj.name .. ":" .. tostring(obj.gen)] = true
        end
    end

    -- Pick thumbnail (determined after bestList is built below)

    -- Helper to format line with duel check
    local function formatLine(obj, bullet)
        local mutStr = obj.mutation and ("  |  " .. obj.mutation) or ""
        local key = obj.name .. ":" .. tostring(obj.gen)
        local isDuel = duelLookup[key] or obj.inDuel or obj.duelCooldown
        if isDuel then
            return string.format(" \xE2\x9A\x94\xEF\xB8\x8F %s [$%s]%s\n", obj.name, formatIncomeNumber(obj.gen), mutStr)
        else
            return string.format(" %s %s [$%s]%s\n", bullet, obj.name, formatIncomeNumber(obj.gen), mutStr)
        end
    end

    -- Merge sorted + same-tier duelItems for Best display
    local bestList = {}
    local bestSeen = {}
    for _, obj in ipairs(sorted) do
        local key = obj.name .. ":" .. tostring(obj.gen)
        if not bestSeen[key] then
            bestSeen[key] = true
            table.insert(bestList, obj)
        end
    end
    if duelItems then
        for _, obj in ipairs(duelItems) do
            local key = obj.name .. ":" .. tostring(obj.gen)
            if not bestSeen[key] then
                bestSeen[key] = true
                table.insert(bestList, obj)
            end
        end
    end
    table.sort(bestList, function(a, b) return a.gen > b.gen end)

    local best = bestList[1]
    for _, obj in ipairs(bestList) do
        if PRIORITY_THUMBNAIL[obj.name] then best = obj; break end
    end

    desc = desc .. "\xF0\x9F\x8F\xB7\xEF\xB8\x8F **Best**\n\n"

    for i = 1, math.min(15, #bestList) do
        desc = desc .. formatLine(bestList[i], "\xE2\x97\x8F")
    end

    -- Append extra items from another range
    if extraItems and #extraItems > 0 then
        local extraSorted = {}
        for _, obj in ipairs(extraItems) do table.insert(extraSorted, obj) end
        table.sort(extraSorted, function(a, b) return a.gen > b.gen end)

        desc = desc .. "\n\xF0\x9F\x8D\x83 **" .. (extraTitle or "Others") .. "**\n\n"
        for i = 1, math.min(15, #extraSorted) do
            desc = desc .. formatLine(extraSorted[i], "\xE2\x80\xA2")
        end
    end

    local playerCount = #Players:GetPlayers()
    local maxPlayers = Players.MaxPlayers
    desc = desc .. "\n\xF0\x9F\xAB\x82 Players " .. playerCount .. "/" .. maxPlayers

    -- Duel timer
    local hasDuel = duelItems and #duelItems > 0
    if hasDuel then
        local duelEndTs = getRealUTC() + 30
        desc = desc .. "\n\xE2\x9A\x94\xEF\xB8\x8F Duel ends: <t:" .. duelEndTs .. ":R>"
    end

    if wh.sendServerInfo then
        desc = desc .. "\n\n\xF0\x9F\x93\x8B **Server**\n```\n" .. tostring(jobId) .. "\n```\n### \xE2\x80\x82\xE2\x80\x82\xE2\x80\x82\xE2\x80\x82\xE2\x80\x82" .. tostring(jobId)
    end

    local fields = {}
    if wh.sendTeleport then
        table.insert(fields, {
            name = "Teleport",
            value = string.format("```lua\ngame:GetService('TeleportService'):TeleportToPlaceInstance(%d, '%s')\n```", placeId, jobId),
            inline = false,
        })
    elseif wh.showJoinerAd then
        table.insert(fields, {
            name = "Joiner",
            value = "[Buy here](https://discord.com/channels/1448597315207299126/1449995006315204891)",
            inline = false,
        })
    end

    local totalFound = #filtered + (extraItems and #extraItems or 0)

    -- Check if any priority brainrot is present for @everyone ping
    -- Only ping if NOT in duel (duel brainrots from duelItems are excluded)
    local hasPriority = false
    for _, obj in ipairs(filtered) do
        if PRIORITY_THUMBNAIL[obj.name] and not (obj.inDuel or obj.duelCooldown) then
            hasPriority = true
            break
        end
    end

    return {
        username = "\xF0\x9F\x8C\xB8 AURORA FINDER | " .. wh.title,
        content = hasPriority and "@everyone" or nil,
        embeds = { {
            description = desc,
            color = wh.color,
            thumbnail = { url = getBrainrotImageUrl(best.name) },
            fields = (#fields > 0) and fields or nil,
            footer = { text = "AuroraNotifier  |  " .. os.date("!%d.%m.%Y %H:%M", getRealUTC()) .. " | " .. BOT_TAG },
        } },
    }
end

local function sendDiscordNotificationByRange(filtered, wh, allowVDS, extraItems, extraTitle, duelItems)
    local req = getRequester()
    if not req or #filtered == 0 then return end

    -- Build payload synchronously (fast)
    local payload = buildDiscordEmbed(filtered, wh, extraItems, extraTitle, duelItems)
    local body = HttpService:JSONEncode(payload)

    -- Send to Discord ASYNC — don't block teleport
    task.spawn(function()
        pcall(function()
            return req({
                Url = wh.url,
                Method = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body = body,
            })
        end)
    end)

    -- VDS also async
    if allowVDS then
        task.spawn(function()
            pcall(function() sendToVDS(filtered, wh, duelItems) end)
        end)
    end
end

local function scanAndNotify()
    local allFound = collectAll(1.0)

    local kingrangeGroup = {}
    local highrangeGroup = {}
    local midrangeGroup = {}
    local lowrangeGroup = {}
    local duelKingGroup = {}
    local duelHighGroup = {}
    local duelMidGroup = {}

    for _, obj in ipairs(allFound) do
        if OBJECTS[obj.name] and shouldShow(obj.name, obj.gen) and type(obj.gen) == "number" then
            -- Active duel = SKIP entirely (don't notify at all)
            if not obj.inDuel then
                local isDuelCD = obj.duelCooldown

                if isLowRange(obj.name, obj.gen) then
                    table.insert(lowrangeGroup, obj)
                else
                    local tier = classifyTier(obj.name, obj.gen)

                    if isDuelCD then
                        if tier == "king" then
                            table.insert(duelKingGroup, obj)
                        elseif tier == "high" then
                            table.insert(duelHighGroup, obj)
                        else
                            table.insert(duelMidGroup, obj)
                        end
                    else
                        if tier == "king" then
                            table.insert(kingrangeGroup, obj)
                        elseif tier == "high" then
                            table.insert(highrangeGroup, obj)
                        else
                            table.insert(midrangeGroup, obj)
                        end
                    end
                end
            end
        end
    end

    local newKing = #kingrangeGroup > 0 and deduplicateForSend(filterNewObjects(kingrangeGroup)) or {}
    local newHigh = #highrangeGroup > 0 and deduplicateForSend(filterNewObjects(highrangeGroup)) or {}
    local newMid  = #midrangeGroup > 0 and deduplicateForSend(filterNewObjects(midrangeGroup)) or {}
    local newDuelKing = #duelKingGroup > 0 and deduplicateForSend(filterNewObjects(duelKingGroup)) or {}
    local newDuelHigh = #duelHighGroup > 0 and deduplicateForSend(filterNewObjects(duelHighGroup)) or {}
    local newDuelMid  = #duelMidGroup > 0 and deduplicateForSend(filterNewObjects(duelMidGroup)) or {}

    -- Duel items per range (only same tier)
    local duelForKing = newDuelKing
    local duelForHigh = newDuelHigh
    local duelForMid = newDuelMid

    if #newKing > 0 then
        -- Set teleport flag IMMEDIATELY so main loop teleports ASAP
        shouldTeleportAfterFind = true
        UpdateStatus("FOUND KING! " .. newKing[1].name, Color3.fromRGB(255, 215, 0))
        -- Others = highrange + midrange combined
        local kingOthers = {}
        for _, obj in ipairs(newHigh) do table.insert(kingOthers, obj) end
        for _, obj in ipairs(newMid) do table.insert(kingOthers, obj) end
        sendDiscordNotificationByRange(newKing, WEBHOOKS[1], true, #kingOthers > 0 and kingOthers or nil, "Others", duelForKing)
        trackFind(newKing, "kingrange")
        if #newHigh > 0 then trackFind(newHigh, "highrange") end
        if #newMid > 0 then trackFind(newMid, "midrange") end
    elseif #newHigh > 0 and #newMid > 0 then
        shouldTeleportAfterFind = true
        UpdateStatus("FOUND HIGH! " .. newHigh[1].name, Color3.fromRGB(120, 255, 120))
        sendDiscordNotificationByRange(newHigh, WEBHOOKS[2], true, newMid, "Others", duelForHigh)
        trackFind(newHigh, "highrange")
        trackFind(newMid, "midrange")
    elseif #newHigh > 0 then
        shouldTeleportAfterFind = true
        UpdateStatus("FOUND HIGH! " .. newHigh[1].name, Color3.fromRGB(120, 255, 120))
        sendDiscordNotificationByRange(newHigh, WEBHOOKS[2], true, nil, nil, duelForHigh)
        trackFind(newHigh, "highrange")
    end

    -- Mid Range webhook (WEBHOOKS[3]) — always send separately
    if #newMid > 0 then
        sendDiscordNotificationByRange(newMid, WEBHOOKS[3], true, nil, nil, duelForMid)
        trackFind(newMid, "midrange")
    end

    -- If only duel brainrots found (no normal ones), still send to appropriate webhook
    if #newKing == 0 and #newHigh == 0 and #newMid == 0 then
        if #newDuelKing > 0 then
            local placeholder = { newDuelKing[1] }
            sendDiscordNotificationByRange(placeholder, WEBHOOKS[1], true, nil, nil, newDuelKing)
        end
        if #newDuelHigh > 0 then
            local placeholder = { newDuelHigh[1] }
            -- Add midrange duels as extraItems (Others)
            local midDuelExtra = #newDuelMid > 0 and newDuelMid or nil
            sendDiscordNotificationByRange(placeholder, WEBHOOKS[2], true, midDuelExtra, "Others", newDuelHigh)
        end
        if #newDuelMid > 0 and #newDuelHigh == 0 then
            local placeholder = { newDuelMid[1] }
            sendDiscordNotificationByRange(placeholder, WEBHOOKS[3], true, nil, nil, newDuelMid)
        end
    end

    -- Low Range webhook (WEBHOOKS[4]) — only if NO king/high/mid found on this server
    local hasAnyMain = #kingrangeGroup > 0 or #highrangeGroup > 0 or #midrangeGroup > 0
        or #duelKingGroup > 0 or #duelHighGroup > 0 or #duelMidGroup > 0
    if not hasAnyMain and #lowrangeGroup > 0 then
        local newLow = deduplicateForSend(filterNewObjects(lowrangeGroup))
        if #newLow > 0 then
            sendDiscordNotificationByRange(newLow, WEBHOOKS[4], false, nil, nil, nil)
        end
    end
end

local function LoadServers()
    local ok, resp = pcall(function() return game:HttpGet(TELEPORT_SETTINGS.PASTEFY_URL) end)
    if not ok then return {} end
    local r = {}
    for id in string.gmatch(resp, "([a-f0-9%-]+)") do table.insert(r, id) end
    return r
end

local function IsServerAvailable(serverId)
    if not BLACKLIST[serverId] then return true end
    return (os.time() - BLACKLIST[serverId]) > TELEPORT_SETTINGS.COOLDOWN_TIME
end

local function TeleportLoop()
    while true do
        local available = {}
        for _, serverId in ipairs(SERVER_LIST) do
            if IsServerAvailable(serverId) then
                table.insert(available, serverId)
            end
        end

        if #available == 0 then
            UpdateStatus("All servers on cooldown, reloading...", Color3.fromRGB(255, 200, 100))
            task.wait(10)
            local fresh = LoadServers()
            if #fresh > 0 then SERVER_LIST = fresh end
            BLACKLIST = {}
        else
            local target = available[math.random(1, #available)]
            UpdateStatus("Connecting: " .. target:sub(1, 8) .. "...", Color3.fromRGB(200, 200, 255))

            local success = pcall(function()
                TeleportService:TeleportToPlaceInstance(TELEPORT_SETTINGS.GAME_ID, target, Players.LocalPlayer)
            end)

            if not success then
                BLACKLIST[target] = os.time()
                UpdateStatus("Failed, retrying...", Color3.fromRGB(255, 150, 150))
                task.wait(TELEPORT_SETTINGS.ERROR_RETRY_DELAY)
            else
                task.wait(TELEPORT_SETTINGS.SUCCESS_DELAY)
            end
        end
    end
end

local COORDINATOR_URL = "http://103.27.157.134:8452"
local SCAN_TIME = 15 -- seconds to scan each server before moving

local function coordinatorRequest(endpoint, body)
    local req = getRequester()
    if not req then return nil end
    local ok, resp = pcall(function()
        return req({
            Url = COORDINATOR_URL .. endpoint,
            Method = body and "POST" or "GET",
            Headers = { ["Content-Type"] = "application/json" },
            Body = body and HttpService:JSONEncode(body) or nil,
        })
    end)
    if ok and resp and resp.StatusCode == 200 then
        local s, data = pcall(function() return HttpService:JSONDecode(resp.Body) end)
        if s then return data end
    end
    return nil
end

local function claimServer(serverId)
    return coordinatorRequest("/claim", { bot = ACCOUNT_NAME, server = serverId })
end

local function sendHeartbeat(serverId)
    return coordinatorRequest("/heartbeat", { bot = ACCOUNT_NAME, server = serverId })
end

local function leaveServer()
    return coordinatorRequest("/leave", { bot = ACCOUNT_NAME })
end

local function getOccupiedServers()
    local data = coordinatorRequest("/occupied")
    if data and data.servers then return data.servers end
    return {}
end

local function fetchServers()
    local servers = {}
    -- Method 1: From VPS servers.txt (GROW collected)
    pcall(function()
        local resp = game:HttpGet(SERVER_URL)
        for id in string.gmatch(resp, "([a-f0-9%-]+)") do
            table.insert(servers, id)
        end
    end)
    -- Method 2: From Roblox API (fallback)
    if #servers == 0 then
        pcall(function()
            local url = "https://games.roblox.com/v1/games/" .. GAME_ID .. "/servers/Public?limit=100"
            local resp = game:HttpGet(url)
            local data = HttpService:JSONDecode(resp)
            if data and data.data then
                for _, s in ipairs(data.data) do
                    if s.id and s.playing and s.playing < s.maxPlayers then
                        table.insert(servers, s.id)
                    end
                end
            end
        end)
    end
    return servers
end

-- Background server list cache (refreshed every 30s)
local _cachedServers = {}
local _cachedOccupied = {}
task.spawn(function()
    while true do
        pcall(function() _cachedServers = fetchServers() end)
        pcall(function() _cachedOccupied = getOccupiedServers() end)
        task.wait(60)
    end
end)

local function findFreeServer()
    -- Use cached lists for instant teleport (no HTTP wait)
    local allServers = _cachedServers
    if #allServers == 0 then
        allServers = fetchServers()
        _cachedServers = allServers
    end
    if #allServers == 0 then return nil end

    local occupied = _cachedOccupied
    if #occupied == 0 then occupied = getOccupiedServers() end
    local occupiedSet = {}
    for _, id in ipairs(occupied) do occupiedSet[id] = true end
    -- Also exclude current server
    occupiedSet[game.JobId] = true

    local free = {}
    for _, id in ipairs(allServers) do
        if not occupiedSet[id] then
            table.insert(free, id)
        end
    end

    if #free > 0 then
        return free[math.random(1, #free)]
    end
    -- All occupied, pick random (avoid current)
    local others = {}
    for _, id in ipairs(allServers) do
        if id ~= game.JobId then
            table.insert(others, id)
        end
    end
    if #others > 0 then
        return others[math.random(1, #others)]
    end
    return nil
end

        leaveServer()

        for attempt = 1, 5 do
            local target = findFreeServer()

                local ok = pcall(function()

                end

                UpdateStatus("No free servers, retry " .. attempt .. "/5", Color3.fromRGB(255, 150, 100))
                task.wait(5)
            end
        end

        if not teleported then
            -- All attempts failed, re-claim current server
            claimServer(game.JobId)
            UpdateStatus("Dup teleport failed, staying", Color3.fromRGB(255, 0, 0))
            task.wait(10)
        end

    if shouldTeleportAfterFind then

        UpdateStatus("Found king/high! Leaving server...", Color3.fromRGB(255, 215, 0))
        leaveServer()

        -- Retry teleport up to 10 times with different servers
        local triedServers = {}
        local teleported = false
        for attempt = 1, 10 do
            local target = findFreeServer()
            -- Avoid retrying same server
            local tries = 0
            while target and triedServers[target] and tries < 5 do
                target = findFreeServer()
                tries = tries + 1
            end
            if not target then
                UpdateStatus("No free servers, retry " .. attempt .. "/10", Color3.fromRGB(255, 150, 100))
                task.wait(2)

                UpdateStatus("Teleport " .. attempt .. "/10: " .. target:sub(1, 8), Color3.fromRGB(200, 200, 255))
                local ok = pcall(function()
                    TeleportService:TeleportToPlaceInstance(GAME_ID, target, localPlayer)
                end)

                    teleported = true
                    task.wait(10)
                    break
                else
                    UpdateStatus("Failed " .. attempt .. "/10, trying another...", Color3.fromRGB(255, 100, 100))
                    task.wait(1)
                end
            end
        end

        if not teleported then
            claimServer(game.JobId)
            UpdateStatus("All teleports failed, staying", Color3.fromRGB(255, 0, 0))
            task.wait(3)
        end
    end

    UpdateStatus("Scans: " .. scanCount .. " | Server: " .. game.JobId:sub(1, 8), Color3.fromRGB(100, 255, 100))
    task.wait(0.1)
end
